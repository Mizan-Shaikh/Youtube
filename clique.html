Clique - HackerRank Solution

This is the C# solution for the Clique problem from HackerRank.

Source Code


using System;

class Solution
{
    static void Main()
    {
        int t = int.Parse(Console.ReadLine());
        for (int test = 0; test < t; test++)
        {
            var tokens = Console.ReadLine().Split();
            int n = int.Parse(tokens[0]);
            int m = int.Parse(tokens[1]);

            int low = 1, high = n;
            int result = n;

            while (low <= high)
            {
                int mid = (low + high) / 2;
                if (MaxEdgesWithoutClique(n, mid) < m)
                {
                    low = mid + 1;
                }
                else
                {
                    result = mid;
                    high = mid - 1;
                }
            }

            Console.WriteLine(result);
        }
    }

    static int MaxEdgesWithoutClique(int n, int r)
    {
        // Using TurÃ¡n's Theorem
        double maxEdges = (1.0 - 1.0 / (r - 1)) * n * n / 2.0;
        return (int)Math.Floor(maxEdges);
    }
}
